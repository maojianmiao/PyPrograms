## Mysql query
create table Topic(
	ID		 	INT UNSIGNED  PRIMARY KEY  AUTO_INCREMENT,
	Title	 	VARCHAR(100)	 NOT NULL,
	Comments	 INT  UNSIGNED	 	NULL,
	Link	 	VARCHAR(100)   	NOT NULL,
	SiteID	 	SMALLINT UNSIGNED ,
	Views	 	INT  UNSIGNED 	NULL,
	Recommends  INT  UNSIGNED  NULL,
	StoryPoints	 	INT	 NULL,
	QulaityPoints 	INT  NULL,
	Guide		 	TEXT  NULL,
	NewsTypeID	 	TINYINT  NULL
)

CREATE TABLE NewsType(
	NewsTypeID 	INT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
	NewsTypeName VARCHAR(20)	NOT NULL,
	Description		TINYTEXT NULL
)

CREATE TABLE Sites(
	SiteID	SMALLINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
	SiteName	VARCHAR(12),
	Link	VARCHAR(30)	
)
